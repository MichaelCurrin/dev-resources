### Sample

```javascript
import { builder } from "@netlify/functions";

const handler = async (event, context) => {
 const content = { message: 'Hello World' };

  return {
    statusCode: 200,
    body: JSON.stringify(content),
  }
}

exports.handler = builder(handler)
```
